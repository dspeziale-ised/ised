function initJobWidgets(jobs) { Object.keys(jobs).forEach(function (name) { const btn = document.querySelector('.job-start-btn[data-job="' + name + '"]'); const stopBtn = document.querySelector('.job-stop-btn[data-job="' + name + '"]'); // querySelectorAll (non querySelector): la stessa badge di stato può // comparire più volte sulla stessa pagina (es. dashboard: sia nella // riga "Azioni rapide" sia nell'indicatore riepilogativo "Scansioni in // corso") - vanno aggiornate tutte, non solo la prima trovata nel DOM. const badges = document.querySelectorAll('.job-status-badge[data-job="' + name + '"]'); const logEl = document.querySelector('.job-log[data-job="' + name + '"]'); const cfg = jobs[name] || {}; function setBadges(running) { badges.forEach(function (badge) { badge.className = 'badge badge-pill job-status-badge ' + (running ? 'badge-info' : 'badge-secondary'); badge.setAttribute('data-job', name); badge.innerHTML = running ? ' in corso' : ' inattivo'; }); } function refreshStatus() { fetch('/api/jobs/' + name + '/status') .then(function (r) { return r.json(); }) .then(function (d) { setBadges(d.running); if (btn) btn.disabled = d.running; if (stopBtn) stopBtn.style.display = d.running ? 'inline-block' : 'none'; if (logEl && typeof d.log === 'string') { const atBottom = logEl.scrollTop + logEl.clientHeight >= logEl.scrollHeight - 10; logEl.textContent = d.log || '(nessun log ancora)'; if (atBottom) logEl.scrollTop = logEl.scrollHeight; } }) .catch(function () {}); } if (btn) { btn.addEventListener('click', function () { btn.disabled = true; const force = cfg.forceCheckbox ? document.getElementById(cfg.forceCheckbox) : null; const body = new URLSearchParams(); if (force && force.checked) body.set('force', '1'); (cfg.fields || []).forEach(function (f) { const el = document.getElementById(f.id); if (el && el.value !== '') body.set(f.param, el.value); }); fetch('/jobs/' + name + '/start', { method: 'POST', body: body }) .then(function (r) { return r.json(); }) .then(function (d) { if (!d.started) { alert(d.reason || 'Non avviato.'); } refreshStatus(); }) .catch(function () { alert('Errore di rete.'); btn.disabled = false; }); }); } if (stopBtn) { stopBtn.addEventListener('click', function () { notifierConfirm( 'Interrompere questa operazione? I dati già salvati restano nel database; il resto del batch corrente andrà rifatto al prossimo avvio.', function () { stopBtn.disabled = true; fetch('/jobs/' + name + '/stop', { method: 'POST' }) .then(function (r) { return r.json(); }) .then(function (d) { if (!d.stopped) { alert(d.reason || 'Non interrotto.'); } stopBtn.disabled = false; refreshStatus(); }) .catch(function () { alert('Errore di rete.'); stopBtn.disabled = false; }); } ); }); } if (badges.length || logEl) { refreshStatus(); setInterval(refreshStatus, 4000); } }); }
